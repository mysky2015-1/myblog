# 8.8.2 高阶函数

    // 高阶函数
    function not(f){
        // not 是个高阶函数
        return function(){
            let result = f.apply(this,arguments)
            // console.log(result)
            return !result
        }
    }
    let even = function(x){
        console.log(x)
        return x % 2 === 0
    }
    let odd = not(even)
    let arr01 = [2,4,6]
    console.log(arr01.every(even)) // 该数组中是否全是偶数
    console.log(arr01.every(odd)) // 该数组中是否全是奇数



  // 高阶函数2
  let  map = Array.prototype.map
      ? function (a,f){ return a.map(f)}
      : function (a,f) {
          let results = []
          for (let i=0,len = a.length;i<len;i++) {
                  if( i in a) results[i] = f.call(null, a[i],i,a)
          }
          return results
      }
  function mapper(f){
    return function(a) {
            return map(a, f)
    }
  }
  let increment = function (x) {
      x = x+1
      return x
  }
  let incrementer = mapper(increment)
  console.log('原方法：',increment(5))
  console.log('高阶函数：',incrementer(arr01))



  // 高阶函数2-1
  // 接受两个函数f()和g() ,并返回一个新的函数用以计算f(g())
  // 返回一个新的可以计算f(g())的函数
  // 返回的函数h()将它所有的实参传入g(),然后将g()的返回值传入f()
  // 调用f()和g()时的this值和调用h()的this值是同一个this
  function compose(f,g){
      return function(){
              return f.call(this,g.apply(this,arguments))
      }
  }
  var square = function(x){ return x*x;}
  var sum = function (x,y){ return x+y }
  var squareSum = compose(square,sum)
  console.log('求和+求积：',squareSum(4,6))

# [数组]

  • 原文：数组是值的有序集合，每个值都叫做数组，而每个元素在数组中有个位置，以数字表示，称为索引；
  • javascript数组是无类型的：
  • 数组元素可以是任意类型，并且同一个数组中的不同元素也可能有不同的类型。
  • 数组的元素甚至也可能是对象或者其它数组，这允许创建复杂的数据结构，比如对象的数组或者数组的数组
  • javascript数组的索引是基于0的32位数值
  • 第一个元素的索引为0，最大可能的索引为 4 294 967 294（2的32次方-2）
  • 数组最大能容纳4 294 967 295个元素
  • javascript数组是动态的
  • 根据需要他们会增长或者缩减，并且在创建数组时无需声明一个固定的大小或者在数组大小变化时无需重新分配空间
  • javascript数组可能是稀疏的，数组元素的索引不一定要连续的，它们之间是可以有空缺的，
  • 每个Javascript数组都有一个length属性，针对非稀疏数组，该属性就是数组元素的个数，针对稀疏数组，length比所有元素的索引都要大
  • javascript数组是javascript对象的特殊形式，数组索引实际上和碰巧是整数的属性名差不多，我们将在本章的其它地方更多的讨论特殊化的数组。通常，数组的实现是要经过优化的，用数字索引来访问数组元素一般来说比访问常规的对象属性要快的很多。
